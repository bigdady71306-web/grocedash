🛠️ Database Setup Required

GroceDash needs a few tables in your Supabase project. Follow these quick steps to enable cloud sync:

1
Go to your Supabase SQL Editor →
2
Copy and paste the SQL below into the editor
3
Click Run, then come back and click "Check Setup" below
-- GroceDash Setup SQL
-- Paste this in Supabase Dashboard → SQL Editor → Run

create table if not exists families (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade unique,
  name text default 'My Family',
  member_count int default 5,
  dietary_notes text,
  weekly_budget numeric default 250,
  store text default 'HEB — N. Frazier, Conroe TX',
  created_at timestamptz default now()
);
alter table families enable row level security;
create policy "Users own family" on families for all using (auth.uid() = user_id);

create table if not exists meal_plans (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade,
  week_start date not null,
  sunday_recipe_id text, monday_recipe_id text,
  tuesday_recipe_id text, wednesday_recipe_id text,
  thursday_recipe_id text, friday_recipe_id text,
  saturday_recipe_id text,
  created_at timestamptz default now(),
  unique(user_id, week_start)
);
alter table meal_plans enable row level security;
create policy "Users own meal_plans" on meal_plans for all using (auth.uid() = user_id);

create table if not exists grocery_lists (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade,
  week_start date not null,
  budget numeric default 250,
  items jsonb default '[]',
  created_at timestamptz default now(),
  unique(user_id, week_start)
);
alter table grocery_lists enable row level security;
create policy "Users own grocery_lists" on grocery_lists for all using (auth.uid() = user_id);

create table if not exists trip_history (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references auth.users(id) on delete cascade,
  trip_date date not null,
  store_name text,
  budget numeric,
  amount_spent numeric,
  item_count int default 0,
  saved numeric,
  created_at timestamptz default now()
);
alter table trip_history enable row level security;
create policy "Users own trip_history" on trip_history for all using (auth.uid() = user_id);
